Demonstration

JavaScript

/*    This is a cool test function
 *    and block comment
 */
function test (arg) {
    var result = null,
        minLen = 0x3;

    console.log(this);

    if (typeof arg === 'undefined') {
        throw new Error('arg is undefined');    // This is an error!
    }

    test = arg.replace(/[,;\s]/g, '\r\n');

    return result.length >= minLen ? result : 'Result is too small!';
}

Using

Place the following code in the HEAD section of your code:

<link id="humpty-theme" rel="stylesheet" type="text/css" href="humpty-light.css" />
<!-- or <link id="humpty-theme" rel="stylesheet" type="text/css" href="humpty-dark.css" /> -->
<script type="text/javascript" src="humpty.js"></script>

Configuration

/* Before "onLoad" event */
Humpty.config({     // <-- this is the default configuration
    autorun         : true,
    hideLineNumbers : false,
    alwaysOpen      : false,
    codeTags        : ['code', 'pre']
});
